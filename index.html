Estruturas Lineares Dinâmicas: Listas, Pilhas e Filas

Painel Teórico

Listas Simples

Estruturas de dados lineares onde os elementos são organizados sequencialmente. Cada nó possui um ponteiro para o próximo elemento da lista.

Inserção flexível Acesso sequencial Tamanho dinâmico

Listas Duplas

Cada nó possui dois ponteiros: um para o próximo elemento e outro para o anterior. Permite navegação bidirecional e operações mais eficientes.

Navegação bidirecional Inserção/remoção eficiente Maior flexibilidade

Listas Circulares

O último elemento aponta para o primeiro, formando um ciclo. Útil para aplicações que requerem acesso cíclico aos dados.

Acesso cíclico Sem início/fim Aplicações especiais

Pilhas (LIFO)

Estrutura que segue o princípio Last In, First Out. O último elemento inserido é o primeiro a ser removido. Operações principais: push (inserir) e pop (remover).

LIFO Push/Pop Recursão

Filas (FIFO)

Estrutura que segue o princípio First In, First Out. O primeiro elemento inserido é o primeiro a ser removido. Operações principais: enqueue (inserir) e dequeue (remover).

FIFO Enqueue/Dequeue Processamento

função inserir(lista, valor, posicao):
    novo_no = criar_no(valor)
    se posicao == 0:
        novo_no.proximo = lista.inicio
        lista.inicio = novo_no
    senão:
        atual = lista.inicio
        para i de 0 até posicao-2:
            atual = atual.proximo
        novo_no.proximo = atual.proximo
        atual.proximo = novo_no
                                
class No:
    def __init__(self, valor):
        self.valor = valor
        self.proximo = None

class Lista:
    def __init__(self):
        self.inicio = None
    
    def inserir(self, valor, posicao=None):
        novo_no = No(valor)
        if posicao == 0 or self.inicio is None:
            novo_no.proximo = self.inicio
            self.inicio = novo_no
        else:
            atual = self.inicio
            for i in range(posicao - 1):
                if atual.proximo:
                    atual = atual.proximo
            novo_no.proximo = atual.proximo
            atual.proximo = novo_no
                                
class No {
    int valor;
    No proximo;
    
    public No(int valor) {
        this.valor = valor;
        this.proximo = null;
    }
}

class Lista {
    private No inicio;
    
    public void inserir(int valor, int posicao) {
        No novoNo = new No(valor);
        if (posicao == 0 || inicio == null) {
            novoNo.proximo = inicio;
            inicio = novoNo;
        } else {
            No atual = inicio;
            for (int i = 0; i < posicao - 1; i++) {
                if (atual.proximo != null) {
                    atual = atual.proximo;
                }
            }
            novoNo.proximo = atual.proximo;
            atual.proximo = novoNo;
        }
    }
}
                                
#include <stdio.h>
#include <stdlib.h>

typedef struct No {
    int valor;
    struct No* proximo;
} No;

typedef struct {
    No* inicio;
} Lista;

void inserir(Lista* lista, int valor, int posicao) {
    No* novo_no = (No*)malloc(sizeof(No));
    novo_no->valor = valor;
    
    if (posicao == 0 || lista->inicio == NULL) {
        novo_no->proximo = lista->inicio;
        lista->inicio = novo_no;
    } else {
        No* atual = lista->inicio;
        for (int i = 0; i < posicao - 1; i++) {
            if (atual->proximo != NULL) {
                atual = atual->proximo;
            }
        }
        novo_no->proximo = atual->proximo;
        atual->proximo = novo_no;
    }
}

D. Simulador de Pilhas

Pilha (LIFO):

Pilha vazia
// Estrutura da pilha
estrutura Pilha:
    itens: array
    topo: inteiro

// Operações básicas
função push(pilha, valor):
    pilha.topo = pilha.topo + 1
    pilha.itens[pilha.topo] = valor

função pop(pilha):
    se pilha.topo >= 0:
        valor = pilha.itens[pilha.topo]
        pilha.topo = pilha.topo - 1
        retorne valor
    senão:
        retorne erro "Pilha vazia"

função peek(pilha):
    se pilha.topo >= 0:
        retorne pilha.itens[pilha.topo]
    senão:
        retorne erro "Pilha vazia"
                                
class Pilha:
    def __init__(self):
        self.itens = []
    
    def push(self, valor):
        self.itens.append(valor)
    
    def pop(self):
        if not self.is_empty():
            return self.itens.pop()
        else:
            raise Exception("Pilha vazia")
    
    def peek(self):
        if not self.is_empty():
            return self.itens[-1]
        else:
            raise Exception("Pilha vazia")
    
    def is_empty(self):
        return len(self.itens) == 0
    
    def size(self):
        return len(self.itens)
                                
import java.util.ArrayList;
import java.util.EmptyStackException;

class Pilha {
    private ArrayList<Integer> itens;
    
    public Pilha() {
        this.itens = new ArrayList<>();
    }
    
    public void push(int valor) {
        itens.add(valor);
    }
    
    public int pop() {
        if (isEmpty()) {
            throw new EmptyStackException();
        }
        return itens.remove(itens.size() - 1);
    }
    
    public int peek() {
        if (isEmpty()) {
            throw new EmptyStackException();
        }
        return itens.get(itens.size() - 1);
    }
    
    public boolean isEmpty() {
        return itens.isEmpty();
    }
    
    public int size() {
        return itens.size();
    }
}
                                
#include <stdio.h>
#include <stdlib.h>

#define MAX_SIZE 100

typedef struct {
    int itens[MAX_SIZE];
    int topo;
} Pilha;

void inicializar(Pilha* pilha) {
    pilha->topo = -1;
}

void push(Pilha* pilha, int valor) {
    if (pilha->topo < MAX_SIZE - 1) {
        pilha->topo++;
        pilha->itens[pilha->topo] = valor;
    } else {
        printf("Pilha cheia\n");
    }
}

int pop(Pilha* pilha) {
    if (pilha->topo >= 0) {
        int valor = pilha->itens[pilha->topo];
        pilha->topo--;
        return valor;
    } else {
        printf("Pilha vazia\n");
        return -1;
    }
}

int peek(Pilha* pilha) {
    if (pilha->topo >= 0) {
        return pilha->itens[pilha->topo];
    } else {
        printf("Pilha vazia\n");
        return -1;
    }
}

E. Simulador de Filas

Fila (FIFO):

Fila vazia
// Estrutura da fila
estrutura Fila:
    itens: array
    inicio: inteiro
    fim: inteiro
    tamanho: inteiro

// Operações básicas
função enqueue(fila, valor):
    se fila.tamanho < MAX_SIZE:
        fila.itens[fila.fim] = valor
        fila.fim = (fila.fim + 1) % MAX_SIZE
        fila.tamanho = fila.tamanho + 1
    senão:
        retorne erro "Fila cheia"

função dequeue(fila):
    se fila.tamanho > 0:
        valor = fila.itens[fila.inicio]
        fila.inicio = (fila.inicio + 1) % MAX_SIZE
        fila.tamanho = fila.tamanho - 1
        retorne valor
    senão:
        retorne erro "Fila vazia"
                                
from collections import deque

class Fila:
    def __init__(self):
        self.itens = deque()
    
    def enqueue(self, valor):
        self.itens.append(valor)
    
    def dequeue(self):
        if not self.is_empty():
            return self.itens.popleft()
        else:
            raise Exception("Fila vazia")
    
    def front(self):
        if not self.is_empty():
            return self.itens[0]
        else:
            raise Exception("Fila vazia")
    
    def is_empty(self):
        return len(self.itens) == 0
    
    def size(self):
        return len(self.itens)
                                
import java.util.LinkedList;
import java.util.Queue;
import java.util.NoSuchElementException;

class Fila {
    private Queue<Integer> itens;
    
    public Fila() {
        this.itens = new LinkedList<>();
    }
    
    public void enqueue(int valor) {
        itens.offer(valor);
    }
    
    public int dequeue() {
        if (isEmpty()) {
            throw new NoSuchElementException("Fila vazia");
        }
        return itens.poll();
    }
    
    public int front() {
        if (isEmpty()) {
            throw new NoSuchElementException("Fila vazia");
        }
        return itens.peek();
    }
    
    public boolean isEmpty() {
        return itens.isEmpty();
    }
    
    public int size() {
        return itens.size();
    }
}
                                
#include <stdio.h>
#include <stdlib.h>

#define MAX_SIZE 100

typedef struct {
    int itens[MAX_SIZE];
    int inicio;
    int fim;
    int tamanho;
} Fila;

void inicializar(Fila* fila) {
    fila->inicio = 0;
    fila->fim = 0;
    fila->tamanho = 0;
}

void enqueue(Fila* fila, int valor) {
    if (fila->tamanho < MAX_SIZE) {
        fila->itens[fila->fim] = valor;
        fila->fim = (fila->fim + 1) % MAX_SIZE;
        fila->tamanho++;
    } else {
        printf("Fila cheia\n");
    }
}

int dequeue(Fila* fila) {
    if (fila->tamanho > 0) {
        int valor = fila->itens[fila->inicio];
        fila->inicio = (fila->inicio + 1) % MAX_SIZE;
        fila->tamanho--;
        return valor;
    } else {
        printf("Fila vazia\n");
        return -1;
    }
}
